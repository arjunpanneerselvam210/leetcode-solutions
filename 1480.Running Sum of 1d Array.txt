class Solution {
    public int[] runningSum(int[] nums) {
        int n = nums.length;
        for(int i = 1; i< n; i++){
            nums[i] = nums[i] + nums[i-1];
        }
        return nums;
    }
}
// computing the sum of current element and the previous element and store it in the current index
// the range starts from 1 to n-1 
// it is a prefix sum approach
